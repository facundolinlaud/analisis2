\section{Conjuntos}
\subsection{Definiciones}
\subsubsection{Compacto}
\teorema{
	\textbf{\underline{Formalización}}: Un conjunto $A \subset \mathbb{R}^n$ es \textbf{compacto} si:
	\begin{itemize}
		\item es cerrado y
		\item es acotado
	\end{itemize}
}

\subsubsection{Arcoconexo}
\teorema{
	\textbf{\underline{Formalización}}: Un conjunto $A \subset \mathbb{R}^n$ es \textbf{arcoconexo} si dados $P,Q \in A$ existe una curva continua $\alpha: [0, 1] \to A$ / $\alpha(0)=P \text{ y } \alpha(1)=Q$.
}
